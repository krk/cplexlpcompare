61  bool LPModel::ReadModel(std::string filename) {
62 
63  if (!boost::filesystem::exists(filename)) {
64  cout << "File not found: " << filename;
65  return false;
66  }
67 
68  boost::iostreams::stream<boost::iostreams::mapped_file_source> file(filename);
69 
70  if (!file){

72  return false;
73  }
74 
75  std::string *line = new std::string();
76  bool reconsider;
77 
78  getline(file, *line);
79  linesRead = 0;
80  while (!file.eof()){
81  reconsider = false;
82 
83  IncLineCount();
84 
85  if (line->length() == 0 || (line->length() > 0 && ((*line)[0] == ' ' || (*line)[0] == '\\'))) {
86  getline(file, *line);
87  continue;
88  }
89 
90  trim(*line);
91 
92  if (boost::iequals(*line, "Generals")
93  || boost::iequals(*line, "General")
94  || boost::iequals(*line, "Gen")) {
95  line = ReadGenerals(file);
96  reconsider = true;
97  }
98  else if (boost::iequals(*line, "Bounds")
99  || boost::iequals(*line, "Bound")) {
100  line = ReadBounds(file);
101  reconsider = true;
102  }
103  else if (boost::iequals(*line, "Binaries")
104  || boost::iequals(*line, "Binary")
105  || boost::iequals(*line, "Bin")) {
106  line = ReadBinaries(file);
107  reconsider = true;
108  }
109  else if (boost::iequals(*line, "SOS")) {
110  line = ReadSosVars(file);
111  reconsider = true;
112  }
113  else if (boost::iequals(*line, "Subject To")
114  || boost::iequals(*line, "such that")
115  || boost::iequals(*line, "st")
116  || boost::iequals(*line, "S.T.")
117  || boost::iequals(*line, "ST.")) {
118  line = ReadConstraints(file);
119  reconsider = true;
120  }
121 
122  if (!reconsider) {
123  getline(file, *line);
124  }
125  }
126 
127  return true;
128  }
