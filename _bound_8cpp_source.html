170  Bound *ret = new Bound();
171 
172  std::vector<std::string> parts;
173  trim(line);
174  ::split(line, ' ', parts, [](const std::string& s) -> bool { return s.length() > 0; });
175 
176  if (parts.size() == 0)
177  return nullptr;
178 
179  // clean this setBounds mess.
180  auto setBounds = [&ret](BoundOp op, std::string s, bool inverted) {
181  auto check_op = inverted ? invert(op) : op;
182 
183  if (op == BoundOp::EQ) {
184  ret->LB = boost::lexical_cast<float>(s);
185  ret->UB = ret->LB;
186  ret->LB_Op = op;
187  ret->UB_Op = op;
188  }
189  else if (op == BoundOp::GT || op == BoundOp::GTE || (check_op == BoundOp::GT || check_op == BoundOp::GTE)) {
190  ret->UB = boost::lexical_cast<float>(s);
191  ret->UB_Op = op;
192  }
193  else if (op == BoundOp::LT || op == BoundOp::LTE || (check_op == BoundOp::LT || check_op == BoundOp::LTE)) {
194  ret->LB = boost::lexical_cast<float>(s);
195  ret->LB_Op = op;
196  }
197  };
198 
199  cmatch what;
200  bool isVarName = regex_match(parts[0].c_str(), what, re_varname);
201 
202  if (isVarName){
203  ret->VarName = parts[0];
204 
205  auto op = get_boundop(parts[1]);
206  if (op != BoundOp::Free)
207  setBounds(invert(op), parts[2], false);
208  }
209  else{
210  auto op = get_boundop(parts[1]);
211  ret->VarName = parts[2];
212 
213  if (op != BoundOp::Free)
214  setBounds(op, parts[0], false);
215 
216  // does have a second part?
217  if (parts.size() == 5)
218  {
219  op = get_boundop(parts[3]);

223  }
